  // Estados para enviar reporte
  const [nuevoPesoReportado, setNuevoPesoReportado] = useState('');
  const [nuevoComentario, setNuevoComentario] = useState('');

  // Estados temporales del formulario de ajuste biométrico
  const [tempBiometrics, setTempBiometrics] = useState({ ...biometrics });
  const [validationWarning, setValidationWarning] = useState("");

  const triggerToast = (message, type = 'success') => {
    setToast({ message, type });
    setTimeout(() => {
      setToast(null);
    }, 4500);
  };

  // Cálculo de Metabolismo Basal y TDEE (Detrás de escenas, sin mostrar calorías a la alumna)
  const metabolismoCalculado = useMemo(() => {
    const { pesoInicial, estatura, edad, actividad, pesoObjetivo } = biometrics;
    
    // Ecuación Mifflin-St Jeor para mujeres
    const bmr = (10 * pesoInicial) + (6.25 * estatura) - (5 * edad) - 161;
    
    // Factor de actividad
    const factores = {
      sedentary: 1.2,
      light: 1.375,
      moderate: 1.55,
      active: 1.725
    };
    const factor = factores[actividad] || 1.375;
    const tdee = bmr * factor;

    // Defecto saludable moderado de déficit (ej. 15% a 20%)
    // Nunca bajamos de un límite seguro de 1350 para mujeres reales
    const targetEnergy = Math.max(1350, Math.round(tdee * 0.82));
    
    // Factor multiplicador de porciones respecto al plan de referencia (1600 kcal)
    const factorPorcion = targetEnergy / 1600;

    // Cálculo de la pérdida total y tasa semanal recomendada
    const totalALoser = pesoInicial - pesoObjetivo;
    const tasaSemanalIdeal = totalALoser / 12; // 12 semanas (90 días)

    return {
      bmr: Math.round(bmr),
      tdee: Math.round(tdee),
      targetEnergy, // Usado solo internamente para cálculo de porciones
      factorPorcion,
      tasaSemanalIdeal: tasaSemanalIdeal.toFixed(2),
      isSafe: tasaSemanalIdeal <= 1.0 // Pérdida saludable es menor o igual a 1.0 kg por semana
    };
  }, [biometrics]);

  // Curva de peso proyectada de forma progresiva semana a semana
  const curvaDePesoProgresiva = useMemo(() => {
    const { pesoInicial, pesoObjetivo } = biometrics;
    const progreso = [];
    const step = (pesoInicial - pesoObjetivo) / 12;
    for (let w = 0; w <= 12; w++) {
      progreso.push({
        semana: w,
        pesoProyectado: (pesoInicial - (step * w)).toFixed(1)
      });
    }
    return progreso;
  }, [biometrics]);

  // Función para escalar los ingredientes de manera dinámica según el metabolismo calculado
  const obtenerIngredientesEscalados = (tipoComida, index) => {
    const comida = OPCIONES_COMIDAS_BASE[tipoComida][index];
    if (!comida) return "";
    
    const factor = metabolismoCalculado.factorPorcion;
    const scaled = {};
    
    // Escalamos cada ingrediente de la base de datos
    Object.keys(comida.ingredientesBase).forEach(key => {
      const val = comida.ingredientesBase[key];
      // Si el ingrediente es un entero pequeño (ej. huevos, claras, rebanadas), redondeamos razonablemente
      if (val < 10) {
        scaled[key] = Math.max(1, Math.round(val * factor * 10) / 10);
      } else {
        // Redondeamos gramos a múltiplos de 5g para practicidad en cocina de mujeres reales
        scaled[key] = Math.round((val * factor) / 5) * 5;
      }
    });

    return comida.formatoIngredientes(scaled);
  };

  const handleSaveBiometrics = (e) => {
    e.preventDefault();
    const pesoDiff = tempBiometrics.pesoInicial - tempBiometrics.pesoObjetivo;
    const tasaSemanal = pesoDiff / 12;

    if (pesoDiff < 0) {
      setValidationWarning("El peso objetivo debe ser menor al peso inicial para el programa de pérdida de peso.");
      return;
    }

    if (tasaSemanal > 1.2) {
      setValidationWarning(`Alerta de Bienestar: Deseas perder ${pesoDiff.toFixed(1)} kg en 90 días. Esto requiere una tasa de ${tasaSemanal.toFixed(2)} kg por semana. Para evitar dietas extremas y proteger tu salud, el sistema regulará tus porciones metabólicas de forma progresiva priorizando un descenso saludable de 0.8 kg semanales.`);
    } else {
      setValidationWarning("");
    }

    setBiometrics({ ...tempBiometrics, isSetupCompleted: true });
    // Ajustar el peso actual de la simulación al nuevo peso inicial configurado
